        <Grid container spacing={24} className={classes.navButtons} justify="center">
          <Grid item xs={12} sm={6} md={4} lg={3}>
            <Button fullWidth variant="contained" color="secondary" onClick={() => this.props.navigation(0)}>
              Reset

    );
